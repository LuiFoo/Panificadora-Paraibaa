# CONFIGURAÇÕES DO BANCO DE DADOS MONGODB
# ============================================
# URI de conexão com o MongoDB
# Para MongoDB Atlas (recomendado para produção):
# MONGODB_URI=mongodb+srv://usuario:senha@cluster.mongodb.net/paraiba?retryWrites=true&w=majority
# Para MongoDB local:
# MONGODB_URI=mongodb://localhost:27017/paraiba
MONGODB_URI=mongodb+srv://usuario:senha@cluster.mongodb.net/paraiba?retryWrites=true&w=majority

# ============================================
# CONFIGURAÇÕES DO NEXTAUTH (AUTENTICAÇÃO)
# ============================================
# URL base da aplicação (sem barra no final)
# Para desenvolvimento local:
# NEXTAUTH_URL=http://localhost:3000
# Para produção:
# NEXTAUTH_URL=https://seudominio.com
NEXTAUTH_URL=http://localhost:3000

# Chave secreta para assinar tokens JWT (use uma string aleatória forte)
# Gere uma chave segura em: https://generate-secret.vercel.app/32
NEXTAUTH_SECRET=sua_chave_secreta_muito_forte_aqui_32_caracteres

# ============================================
# CONFIGURAÇÕES DO GOOGLE OAUTH
# ============================================
# Para obter essas credenciais:
# 1. Acesse: https://console.developers.google.com/
# 2. Crie um novo projeto ou selecione um existente
# 3. Ative a API "Google+ API" e "Google OAuth2 API"
# 4. Vá em "Credenciais" > "Criar credenciais" > "ID do cliente OAuth 2.0"
# 5. Configure as URLs autorizadas:
#    - Origens JavaScript autorizadas: http://localhost:3000 (desenvolvimento)
#    - URLs de redirecionamento autorizadas: http://localhost:3000/api/auth/callback/google
# 6. Para produção, adicione também:
#    - Origens JavaScript autorizadas: https://seudominio.com
#    - URLs de redirecionamento autorizadas: https://seudominio.com/api/auth/callback/google

# ID do cliente Google OAuth
GOOGLE_CLIENT_ID=seu_google_client_id_aqui.apps.googleusercontent.com

# Chave secreta do cliente Google OAuth
GOOGLE_CLIENT_SECRET=sua_google_client_secret_aqui

# ============================================
# CONFIGURAÇÕES DE SEGURANÇA JWT
# ============================================
# Chave secreta para assinar tokens JWT (diferente do NEXTAUTH_SECRET)
# Use uma string aleatória forte e única
# Gere uma chave segura em: https://generate-secret.vercel.app/32
JWT_SECRET=sua_chave_jwt_secreta_muito_forte_aqui_32_caracteres

# ============================================
# CONFIGURAÇÕES DE AMBIENTE
# ============================================
# Define o ambiente da aplicação
# Valores possíveis: development, production, test
NODE_ENV=development
